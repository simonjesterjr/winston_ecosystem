# EODHD API token placement (NOT the API URL)
#
# Copy to eodhd.env (never commit the real token).
#
# Your api_token from https://eodhd.com — looks like: 6a24509831ebb1.76963503
# EODHD_API_KEY=your_token_here
#
# Optional alias for commodity downloads (same token is fine for most plans):
# EODHD_COMMODITY_API_KEY=your_token_here
#
# WRONG — do not paste endpoint URLs as the key:
# EODHD_COMMODITY_API_KEY=https://eodhd.com/api/commodities/historical/{CODE}
#
# Commodity FRED API endpoint (DM calls this for you):
#   GET https://eodhd.com/api/commodities/historical/WTI?api_token=YOUR_TOKEN&interval=daily
#
# Commodity ETF API (standard EOD):
#   GET https://eodhd.com/api/eod/GLD.US?api_token=YOUR_TOKEN
#
# Local dev URLs:
# DATA_MANAGER_URL=http://localhost:3001
# WUT_BASE_URL=http://localhost:3000
